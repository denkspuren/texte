boolean isGreater(int i, int j) { // 2
    assert i >= 0 && j >= 0;      // assert nicht gefordert
    if (i == 0) return false;     // 2
    if (j == 0) return true;      // 4
    return isGreater(--i, --j);   // 2
}

Punktdistanz (20 Punkte)

import java.lang.Math; // 2

class Point {
    private double x, y;         // 2
    Point(double x, double y) {  // 2
        this.x = x;              // 2
        this.y = y;              // 2
    }
    public double dist(Point p) { // 2
        assert p != null;         // nicht gefordert
        return Math.sqrt(Math.pow(x - p.x,2) + Math.pow(y - p.y,2)); // 4
    }
    public String toString() {           // 2
        return "P(" + x + "," + y + ")"; // 2
    }
}

Schranke implementieren (20 Punkte)

class Gate implements Passable { // 5 für Kopf und Variablen
    public final int capacity;
    private Set<Ticket> tickets = new HashSet<>();

    public Gate(int capacity) { // 5
        assert capacity >= 0 : "specify appropriate capacity";
        this.capacity = capacity;
    }

    public Ticket checkIn() throws GateException { // 5 
        if (tickets.size() >= capacity) 
            throw new GateException("capacity exceeded"); 
        Ticket t = new Ticket(); 
        tickets.add(t); 
        return t; 
    }

    public boolean checkOut(Ticket t) { // 5 zusammen mit `remove`-Angabe
    // `boolean remove(Object o)` ist benötigte Methode
        return tickets.remove(t);
    } 
}

Interface aufstellen und nutzen (7 Punkte)

interface Passable {
    Ticket checkIn() throws GateException; // -2 wenn `throws` fehlt
    boolean checkOut(Ticket t);
}

boolean isPalindrome(String s) {
    char[] word = s.toLowerCase().toCharArray();
    int i = 0, j = word.length - 1;
    while(i <= j) { if (word[i++] != word[j--]) return false; }
    return true;
}
